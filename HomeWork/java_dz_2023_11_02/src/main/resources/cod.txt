package org.example;

public class Bicycle {
    private String model;

    private class Handlebar {
        private int strength;
        private boolean isBroken;

        public Handlebar(int strength) {
            this.strength = strength;
            if (strength > 0) isBroken = true;
        }

        public void use(int damage) {
            if (isBroken) {
                if ((strength - damage) <= 0) {
                    isBroken = false;
                    System.out.println("Руль сломан!");
                }
            }
        }
    }

    private class Saddle {
